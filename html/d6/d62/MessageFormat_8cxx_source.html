59 int MessageFormat::addMessage(AliHLTUInt8_t* buffer, unsigned size)
60 {
61  // add message
62  // this will extract the block descriptors from the message
63  // the descriptors refer to data in the original message buffer
64 
65  unsigned count = mBlockDescriptors.size();
66  // the buffer might start with an event descriptor of type AliHLTComponentEventData
67  // the first read attempt is assuming the descriptor and checking consistency
68  // - buffer size at least size of AliHLTComponentEventData struct
69  // - fStructSize member matches
70  // - number of blocks matches fBlockCnt member
71  unsigned position=0;
72  AliHLTComponentEventData* evtData=reinterpret_cast<AliHLTComponentEventData*>(buffer+position);
73  if (position+sizeof(AliHLTComponentEventData)<=size &&
74  evtData->fStructSize==sizeof(AliHLTComponentEventData)) {
75  position += sizeof(AliHLTComponentEventData);
76  } else {
77  // one of the criteria does not match -> no event descriptor
78  evtData=NULL;
79  }
80  do {
81  if (evtData && evtData->fBlockCnt==0 && size<sizeof(AliHLTComponentBlockData)) {
82  // special case: no block data, only event header
83  break;
84  }
85  if (readBlockSequence(buffer+position, size-position, mBlockDescriptors) < 0 ||
86  evtData!=NULL && ((mBlockDescriptors.size()-count) != evtData->fBlockCnt)) {
87  // not in the format of a single block, check if its a HOMER block
88  if (readHOMERFormat(buffer+position, size-position, mBlockDescriptors) < 0 ||
89  evtData!=NULL && ((mBlockDescriptors.size()-count) != evtData->fBlockCnt)) {
90  // not in HOMER format either
91  if (position>0) {
92  // try once more without the assumption of event data header
93  position=0;
94  evtData=NULL;
95  continue;
96  }
97  return -ENODATA;
98  }
99  }
100  } while (0);
101 
102  int result=0;
103  if (evtData && (result=insertEvtData(*evtData))<0) {
104  // error in the event data header, probably headers of different events
105  mBlockDescriptors.resize(count);
106  return result;
107  }
108 
109  return mBlockDescriptors.size() - count;
110 }

132 int MessageFormat::readBlockSequence(AliHLTUInt8_t* buffer, unsigned size,
133  vector<AliHLTComponentBlockData>& descriptorList) const
134 {
135  // read a sequence of blocks consisting of AliHLTComponentBlockData followed by payload
136  // from a buffer
137  if (buffer == NULL) return 0;
138  unsigned position = 0;
139  vector<AliHLTComponentBlockData> input;
140  while (position + sizeof(AliHLTComponentBlockData) < size) {
141  AliHLTComponentBlockData* p = reinterpret_cast<AliHLTComponentBlockData*>(buffer + position);
142  if (p->fStructSize == 0 || // no valid header
143  p->fStructSize + position > size || // no space for the header
144  p->fStructSize + p->fSize + position > size) { // no space for the payload
145  // the buffer is only a valid sequence of data blocks if payload
146  // of the last block exacly matches the buffer boundary
147  // otherwize all blocks added until now are ignored
148  return -ENODATA;
149  }
150  // insert a new block
151  input.push_back(*p);
152  position += p->fStructSize;
153  if (p->fSize > 0) {
154  input.back().fPtr = buffer + position;
155  position += p->fSize;
156  } else {
157  // Note: also a valid block, payload is optional
158  input.back().fPtr = NULL;
159  }
160  // offset always 0 for iput blocks
161  input.back().fOffset = 0;
162  }
163 
164  descriptorList.insert(descriptorList.end(), input.begin(), input.end());
165  return input.size();
166 }

168 int MessageFormat::readHOMERFormat(AliHLTUInt8_t* buffer, unsigned size,
169  vector<AliHLTComponentBlockData>& descriptorList) const
170 {
171  // read message payload in HOMER format
172  if (mpFactory == NULL) const_cast<MessageFormat*>(this)->mpFactory = new ALICE::HLT::HOMERFactory;
173  if (buffer == NULL || mpFactory == NULL) return -EINVAL;
174  unique_ptr<AliHLTHOMERReader> reader(mpFactory->OpenReaderBuffer(buffer, size));
175  if (reader.get() == NULL) return -ENOMEM;
176 
177  unsigned nofBlocks = 0;
178  if (reader->ReadNextEvent() == 0) {
179  nofBlocks = reader->GetBlockCnt();
180  for (unsigned i = 0; i < nofBlocks; i++) {
181  AliHLTComponentBlockData block;
182  memset(&block, 0, sizeof(AliHLTComponentBlockData));
183  block.fStructSize = sizeof(AliHLTComponentBlockData);
184  block.fDataType.fStructSize = sizeof(AliHLTComponentDataType);
185  homer_uint64 id = byteSwap64(reader->GetBlockDataType(i));
186  homer_uint32 origin = byteSwap32(reader->GetBlockDataOrigin(i));
187  memcpy(&block.fDataType.fID, &id,
188  sizeof(id) > kAliHLTComponentDataTypefIDsize ? kAliHLTComponentDataTypefIDsize : sizeof(id));
189  memcpy(&block.fDataType.fOrigin, &origin,
190  sizeof(origin) > kAliHLTComponentDataTypefOriginSize ? kAliHLTComponentDataTypefOriginSize : sizeof(origin));
191  block.fSpecification = reader->GetBlockDataSpec(i);
192  block.fPtr = const_cast<void*>(reader->GetBlockData(i));
193  block.fSize = reader->GetBlockDataLength(i);
194  descriptorList.push_back(block);
195  }
196  }
197 
198  return nofBlocks;
199 }

201 vector<MessageFormat::BufferDesc_t> MessageFormat::createMessages(const AliHLTComponentBlockData* blocks,
202  unsigned count, unsigned totalPayloadSize,
203  const AliHLTComponentEventData& evtData,
204  boost::signals2::signal<unsigned char* (unsigned int)> *cbAllocate)
205 {
206  const AliHLTComponentBlockData* pOutputBlocks = blocks;
207  AliHLTUInt32_t outputBlockCnt = count;
208  mDataBuffer.clear();
209  mMessages.clear();
210  if (mOutputMode == kOutputModeHOMER) {
211  AliHLTHOMERWriter* pWriter = createHOMERFormat(pOutputBlocks, outputBlockCnt);
212  if (pWriter) {
213  AliHLTUInt32_t position = mDataBuffer.size();
214  AliHLTUInt32_t offset = 0;
215  AliHLTUInt32_t payloadSize = pWriter->GetTotalMemorySize();
216  auto msgSize=payloadSize + sizeof(evtData);
217  auto pTarget=&mDataBuffer[position];
218  if (cbAllocate==nullptr) {
219  // make the target in the internal buffer
220  mDataBuffer.resize(position + msgSize);
221  } else {
222  // use callback to create target
223  pTarget=*(*cbAllocate)(msgSize);
224  if (pTarget==nullptr) {
225  throw std::bad_alloc();
226  }
227  }
228  memcpy(pTarget + offset, &evtData, sizeof(evtData));
229  offset+=sizeof(evtData);
230  pWriter->Copy(pTarget + offset, 0, 0, 0, 0);
231  mpFactory->DeleteWriter(pWriter);
232  offset+=payloadSize;
233  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
234  }
235  } else if (mOutputMode == kOutputModeMultiPart || mOutputMode == kOutputModeSequence) {
236  // the output blocks are assempled in the internal buffer, for each
237  // block BlockData is added as header information, directly followed
238  // by the block payload
239  //
240  // kOutputModeMultiPart:
241  // multi part mode adds one buffer descriptor per output block
242  // the devices decides what to do with the multiple descriptors, one
243  // option is to send them in a multi-part message
244  //
245  // kOutputModeSequence:
246  // sequence mode concatenates the output blocks in the internal
247  // buffer. In contrast to multi part mode, only one buffer descriptor
248  // for the complete sequence is handed over to device
249  AliHLTUInt32_t position = mDataBuffer.size();
250  auto pTarget=&mDataBuffer[position];
251  AliHLTUInt32_t offset = 0;
252  unsigned bi = 0;
253  const auto* pOutputBlock = pOutputBlocks;
254  auto maxBufferSize = sizeof(evtData) + count * sizeof(AliHLTComponentBlockData) + totalPayloadSize;
255  do {
256  if (bi == 0 ||
257  mOutputMode == kOutputModeMultiPart) {
258  // request a new message buffer when entering for the first time in concatanate mode
259  // and for every block in multi part mode
260  // the actual size depends on mode and block index
261  // - event data structure is only written at beginning of first
262  // message, regardsless of mode
263  // - concatanate mode requests one big buffer for sequential sequence of
264  // all blocks
265  auto msgSize = (bi==0?sizeof(evtData):0); // first message has event data
266  if (count>0 && mOutputMode==kOutputModeMultiPart) {
267  msgSize+=sizeof(AliHLTComponentBlockData) + pOutputBlock->fSize;
268  } else {
269  msgSize+=count*sizeof(AliHLTComponentBlockData) + totalPayloadSize;
270  }
271  if (cbAllocate==nullptr) {
272  // make the target in the internal buffer, for simplicity data is copied
273  // to a new buffer, a memmove would be possible to make room for block
274  // descriptors
275  if (mDataBuffer.size() < position + msgSize) {
276  if (bi != 0) {
277  throw std::runtime_error("complete buffer allocation must be done in the first cycle");
278  }
279  // resize to the full size at once
280  mDataBuffer.resize(position + maxBufferSize);
281  }
282  pTarget=&mDataBuffer[position];
283  } else {
284  // use callback to create target
285  pTarget=*(*cbAllocate)(msgSize);
286  if (pTarget==nullptr) {
287  throw std::bad_alloc();
288  }
289  }
290  offset=0;
291  }
292 
293  if (bi==0) {
294  // event data only in the first message in order to avoid increase of required
295  // buffer size due to duplicated event header
296  memcpy(pTarget + offset, &evtData, sizeof(evtData));
297  if (mOutputMode == kOutputModeMultiPart && evtData.fBlockCnt>1) {
298  // in multipart mode, there is only one block per part
299  // consequently, the number of blocks indicated in the event data header
300  // does not reflect the number of blocks in this data sample. But it is
301  // set to 1 to make the single message consistent
302  auto* pEvtData = reinterpret_cast<AliHLTComponentEventData*>(pTarget + offset);
303  pEvtData->fBlockCnt=1;
304  }
305  offset+=sizeof(evtData);
306  }
307 
308  if (bi<count) {
309  // copy BlockData and payload
310  AliHLTUInt8_t* pData = reinterpret_cast<AliHLTUInt8_t*>(pOutputBlock->fPtr);
311  pData += pOutputBlock->fOffset;
312  auto* bdTarget = reinterpret_cast<AliHLTComponentBlockData*>(pTarget + offset);
313  memcpy(bdTarget, pOutputBlock, sizeof(AliHLTComponentBlockData));
314  bdTarget->fOffset = 0;
315  bdTarget->fPtr = NULL;
316  offset += sizeof(AliHLTComponentBlockData);
317  memcpy(pTarget + offset, pData, pOutputBlock->fSize);
318  offset += pOutputBlock->fSize;
319  if (mOutputMode == kOutputModeMultiPart) {
320  // send one descriptor per block back to device
321  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
322  position+=offset;
323  }
324  pOutputBlock++;
325  }
326  }
327  while (++bi<count);
328  if (mOutputMode == kOutputModeSequence || count==0) {
329  // send one single descriptor for all concatenated blocks
330  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
331  }
332  } else {
333  // invalid output mode
334  cerr << "error ALICE::HLT::Component: invalid output mode " << mOutputMode << endl;
335  }
336  return mMessages;
337 }

339 AliHLTHOMERWriter* MessageFormat::createHOMERFormat(const AliHLTComponentBlockData* pOutputBlocks,
340  AliHLTUInt32_t outputBlockCnt) const
341 {
342  // send data blocks in HOMER format in one message
343  int iResult = 0;
344  if (mpFactory == NULL) const_cast<MessageFormat*>(this)->mpFactory = new ALICE::HLT::HOMERFactory;
345  if (!mpFactory) return NULL;
346  unique_ptr<AliHLTHOMERWriter> writer(mpFactory->OpenWriter());
347  if (writer.get() == NULL) return NULL;
348 
349  homer_uint64 homerHeader[kCount_64b_Words];
350  HOMERBlockDescriptor homerDescriptor(homerHeader);
351 
352  const AliHLTComponentBlockData* pOutputBlock = pOutputBlocks;
353  for (unsigned blockIndex = 0; blockIndex < outputBlockCnt; blockIndex++, pOutputBlock++) {
354  if (pOutputBlock->fPtr == NULL && pOutputBlock->fSize > 0) {
355  cerr << "warning: ignoring block " << blockIndex << " because of missing data pointer" << endl;
356  }
357  memset(homerHeader, 0, sizeof(homer_uint64) * kCount_64b_Words);
358  homerDescriptor.Initialize();
359  homer_uint64 id = 0;
360  homer_uint64 origin = 0;
361  memcpy(&id, pOutputBlock->fDataType.fID, sizeof(homer_uint64));
362  memcpy(((AliHLTUInt8_t*)&origin) + sizeof(homer_uint32), pOutputBlock->fDataType.fOrigin, sizeof(homer_uint32));
363  homerDescriptor.SetType(byteSwap64(id));
364  homerDescriptor.SetSubType1(byteSwap64(origin));
365  homerDescriptor.SetSubType2(pOutputBlock->fSpecification);
366  homerDescriptor.SetBlockSize(pOutputBlock->fSize);
367  writer->AddBlock(homerHeader, pOutputBlock->fPtr);
368  }
369  return writer.release();
370 }

372 int MessageFormat::insertEvtData(const AliHLTComponentEventData& evtData)
373 {
374  // insert event header to list, sort by time, oldest first
375  if (mListEvtData.size()==0) {
376  mListEvtData.push_back(evtData);
377  } else {
378  vector<AliHLTComponentEventData>::iterator it=mListEvtData.begin();
379  for (; it!=mListEvtData.end(); it++) {
380  if ((it->fEventCreation_us*1e3 + it->fEventCreation_us/1e3)>
381  (evtData.fEventCreation_us*1e3 + evtData.fEventCreation_us/1e3)) {
382  // found a younger element
383  break;
384  }
385  }
386  // TODO: simple logic at the moment, header is not inserted
387  // if there is a mismatch, as the headers are inserted one by one, all
388  // headers in the list have the same ID
389  if (it != mListEvtData.end() &&
390  evtData.fEventID!=it->fEventID) {
391  cerr << "Error: mismatching event ID " << evtData.fEventID
392  << ", expected " << it->fEventID
393  << " for event with timestamp "
394  << evtData.fEventCreation_us*1e3 + evtData.fEventCreation_us/1e3 << " ms"
395  << endl;
396  return -1;
397  }
398  // insert before the younger element
399  mListEvtData.insert(it, evtData);
400  }
401  return 0;
402 }

404 AliHLTUInt64_t MessageFormat::byteSwap64(AliHLTUInt64_t src) const
405 {
406  // swap a 64 bit number
407  return ((src & 0xFFULL) << 56) |
408  ((src & 0xFF00ULL) << 40) |
409  ((src & 0xFF0000ULL) << 24) |
410  ((src & 0xFF000000ULL) << 8) |
411  ((src & 0xFF00000000ULL) >> 8) |
412  ((src & 0xFF0000000000ULL) >> 24) |
413  ((src & 0xFF000000000000ULL) >> 40) |
414  ((src & 0xFF00000000000000ULL) >> 56);
