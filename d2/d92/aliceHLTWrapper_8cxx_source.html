503 int preprocessSocketsDDS(vector<SocketProperties_t>& sockets, std::string networkPrefix)
504 {
505  // binding sockets
506  // - required arguments: size, type, property name, min, max port
507  // - find available port within given range
508  // - create address
509  // - send property to DDS
510  //
511  // connecting sockets
512  // - required arguments: size, property name, count
513  // - replicate according to count
514  // - fetch property from DDS
515  // - if local, only treat properties from tasks on the same node
516  // - translate type from type of opposite socket
517  // - add address
518  int iResult=0;
519  vector<SocketProperties_t> ddsduplicates;
520  for (vector<SocketProperties_t>::iterator sit=sockets.begin();
521  sit!=sockets.end(); sit++) {
522  if (sit->method.compare("bind")==0) {
523  unsigned maskRequiredParams=(0x1<<SIZE)|(0x1<<TYPE)|(0x1<<PROPERTY)|(0x1<<MINPORT);
524  if ((sit->validParams&maskRequiredParams)!=maskRequiredParams) {
525  cerr << buildSocketParameterErrorMsg(maskRequiredParams, sit->validParams, "Error: missing parameter(s) for binding socket") << endl;
526  iResult=-1;
527  break;
528  }
529  // the port will be selected by the FairMQ framework during the
530  // bind process, the address is a placeholder at the moment
531  sit->address="tcp://"+networkPrefix+":";
532  } else if (sit->method.compare("connect")==0) {
533  unsigned maskRequiredParams=(0x1<<SIZE)|(0x1<<PROPERTY)|(0x1<<COUNT);
534  if ((sit->validParams&maskRequiredParams)!=maskRequiredParams) {
535  cerr << buildSocketParameterErrorMsg(maskRequiredParams, sit->validParams, "Error: missing parameter(s) for connecting socket") << endl;
536  iResult=-1;
537  break;
538  }
539  // the port adress will be read from DDS properties
540  // address initialized to the host ip/name which is then used to match
541  // properties on the same hostwhen using local mode
542  // the actual number of input ports is spefified by the count argument, the
543  // corresponding number of properties is expected from DDS
544  sit->address=networkPrefix;
545  // add n-1 duplicates of the port configuration
546  for (int i=0; i<sit->ddscount-1; i++) {
547  ddsduplicates.push_back(*sit);
548  ddsduplicates.back().ddscount=0;
549  }
550  } else {
551  cerr << "Error: invalid socket method '" << sit->method << "'" << endl;
552  iResult=-1; // TODO: find error codes
553  break;
554  }
555  }
556 
557  if (iResult>=0) {
558  sockets.insert(sockets.end(), ddsduplicates.begin(), ddsduplicates.end());
559  }
560  return iResult;
561 }

588 int readSocketPropertiesDDS(vector<SocketProperties_t>& sockets)
589 {
590  // read dds properties for connecting sockets
591  for (vector<SocketProperties_t>::iterator sit=sockets.begin();
592  sit!=sockets.end(); sit++) {
593  if (sit->method.compare("connect")==1) continue;
594  if (sit->ddscount==0) continue; // the previously inserted duplicates
595 
596 #ifdef ENABLE_DDS
597  dds::CKeyValue ddsKeyValue;
598  dds::CKeyValue::valuesMap_t values;
599 
600  std::string hostaddress=sit->address;
601  vector<SocketProperties_t>::iterator workit=sit;
