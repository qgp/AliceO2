275 void MagneticField::MachineField(const Double_t* x, Double_t* b) const
276 {
277  // ---- This is the ZDC part
278  // Compansators for Alice Muon Arm Dipole
279  const Double_t kBComp1CZ = 1075., kBComp1hDZ = 260. / 2., kBComp1SqR = 4.0 * 4.0;
280  const Double_t kBComp2CZ = 2049., kBComp2hDZ = 153. / 2., kBComp2SqR = 4.5 * 4.5;
281 
282  const Double_t kTripQ1CZ = 2615., kTripQ1hDZ = 637. / 2., kTripQ1SqR = 3.5 * 3.5;
283  const Double_t kTripQ2CZ = 3480., kTripQ2hDZ = 550. / 2., kTripQ2SqR = 3.5 * 3.5;
284  const Double_t kTripQ3CZ = 4130., kTripQ3hDZ = 550. / 2., kTripQ3SqR = 3.5 * 3.5;
285  const Double_t kTripQ4CZ = 5015., kTripQ4hDZ = 637. / 2., kTripQ4SqR = 3.5 * 3.5;
286 
287  const Double_t kDip1CZ = 6310.8, kDip1hDZ = 945. / 2., kDip1SqRC = 4.5 * 4.5, kDip1SqRA = 3.375 * 3.375;
288  const Double_t kDip2CZ = 12640.3, kDip2hDZ = 945. / 2., kDip2SqRC = 4.5 * 4.5, kDip2SqRA = 3.75 * 3.75;
289  const Double_t kDip2DXC = 9.7, kDip2DXA = 9.4;
290 
291  double rad2 = x[0] * x[0] + x[1] * x[1];
292 
293  b[0] = b[1] = b[2] = 0;
294 
295  // SIDE C
296  if (x[2] < 0.) {
297  if (TMath::Abs(x[2] + kBComp2CZ) < kBComp2hDZ && rad2 < kBComp2SqR) {
298  b[0] = mCompensatorField2C * mMultipicativeFactorDipole;
299  } else if (TMath::Abs(x[2] + kTripQ1CZ) < kTripQ1hDZ && rad2 < kTripQ1SqR) {
300  b[0] = mQuadrupoleGradient * x[1];
301  b[1] = mQuadrupoleGradient * x[0];
302  } else if (TMath::Abs(x[2] + kTripQ2CZ) < kTripQ2hDZ && rad2 < kTripQ2SqR) {
303  b[0] = -mQuadrupoleGradient * x[1];
304  b[1] = -mQuadrupoleGradient * x[0];
305  } else if (TMath::Abs(x[2] + kTripQ3CZ) < kTripQ3hDZ && rad2 < kTripQ3SqR) {
306  b[0] = -mQuadrupoleGradient * x[1];
307  b[1] = -mQuadrupoleGradient * x[0];
308  } else if (TMath::Abs(x[2] + kTripQ4CZ) < kTripQ4hDZ && rad2 < kTripQ4SqR) {
309  b[0] = mQuadrupoleGradient * x[1];
310  b[1] = mQuadrupoleGradient * x[0];
311  } else if (TMath::Abs(x[2] + kDip1CZ) < kDip1hDZ && rad2 < kDip1SqRC) {
312  b[1] = mDipoleField;
313  } else if (TMath::Abs(x[2] + kDip2CZ) < kDip2hDZ && rad2 < kDip2SqRC) {
314  double dxabs = TMath::Abs(x[0]) - kDip2DXC;
315  if ((dxabs * dxabs + x[1] * x[1]) < kDip2SqRC) {
316  b[1] = -mDipoleField;
317  }
318  }
319  }
320 
321  // SIDE A
322  else {
323  if (TMath::Abs(x[2] - kBComp1CZ) < kBComp1hDZ && rad2 < kBComp1SqR) {
324  // Compensator magnet at z = 1075 m
325  b[0] = mCompensatorField1A * mMultipicativeFactorDipole;
326  }
327 
328  if (TMath::Abs(x[2] - kBComp2CZ) < kBComp2hDZ && rad2 < kBComp2SqR) {
329  b[0] = mCompensatorField2A * mMultipicativeFactorDipole;
330  } else if (TMath::Abs(x[2] - kTripQ1CZ) < kTripQ1hDZ && rad2 < kTripQ1SqR) {
331  b[0] = -mQuadrupoleGradient * x[1];
332  b[1] = -mQuadrupoleGradient * x[0];
333  } else if (TMath::Abs(x[2] - kTripQ2CZ) < kTripQ2hDZ && rad2 < kTripQ2SqR) {
334  b[0] = mQuadrupoleGradient * x[1];
335  b[1] = mQuadrupoleGradient * x[0];
336  } else if (TMath::Abs(x[2] - kTripQ3CZ) < kTripQ3hDZ && rad2 < kTripQ3SqR) {
337  b[0] = mQuadrupoleGradient * x[1];
338  b[1] = mQuadrupoleGradient * x[0];
339  } else if (TMath::Abs(x[2] - kTripQ4CZ) < kTripQ4hDZ && rad2 < kTripQ4SqR) {
340  b[0] = -mQuadrupoleGradient * x[1];
341  b[1] = -mQuadrupoleGradient * x[0];
342  } else if (TMath::Abs(x[2] - kDip1CZ) < kDip1hDZ && rad2 < kDip1SqRA) {
343  b[1] = -mDipoleField;
344  } else if (TMath::Abs(x[2] - kDip2CZ) < kDip2hDZ && rad2 < kDip2SqRA) {
345  double dxabs = TMath::Abs(x[0]) - kDip2DXA;
346  if ((dxabs * dxabs + x[1] * x[1]) < kDip2SqRA) {
347  b[1] = mDipoleField;
348  }
349  }
350  }

447 MagneticField* MagneticField::createFieldMap(Float_t l3Cur, Float_t diCur, Int_t convention, Bool_t uniform,
448  Float_t beamenergy, const Char_t* beamtype, const Char_t* path)
449 {
450  const Float_t l3NominalCurrent1 = 30000.; // (A)
451  const Float_t l3NominalCurrent2 = 12000.; // (A)
452  const Float_t diNominalCurrent = 6000.; // (A)
453 
454  const Float_t tolerance = 0.03; // relative current tolerance
455  const Float_t zero = 77.; // "zero" current (A)
456 
457  BMap_t map = k5kG;
458  double sclL3, sclDip;
459 
460  Float_t l3Pol = l3Cur > 0 ? 1 : -1;
461  Float_t diPol = diCur > 0 ? 1 : -1;
462 
463  l3Cur = TMath::Abs(l3Cur);
464  diCur = TMath::Abs(diCur);
465 
466  if (TMath::Abs((sclDip = diCur / diNominalCurrent) - 1.) > tolerance && !uniform) {
467  if (diCur <= zero) {
468  sclDip = 0.; // some small current.. -> Dipole OFF
469  } else {
470  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN, "Wrong dipole current (%f A)!", diCur);
471  }
472  }
473 
474  if (uniform) {
475  // special treatment of special MC with uniform mag field (normalized to 0.5 T)
476  // no check for scaling/polarities are done
477  map = k5kGUniform;
478  sclL3 = l3Cur / l3NominalCurrent1;
479  } else {
480  if (TMath::Abs((sclL3 = l3Cur / l3NominalCurrent1) - 1.) < tolerance) {
481  map = k5kG;
482  } else if (TMath::Abs((sclL3 = l3Cur / l3NominalCurrent2) - 1.) < tolerance) {
483  map = k2kG;
484  } else if (l3Cur <= zero && diCur <= zero) {
485  sclL3 = 0;
486  sclDip = 0;
487  map = k5kGUniform;
488  } else {
489  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN, "Wrong L3 current (%f A)!", l3Cur);
490  }
491  }
492 
493  if (sclDip != 0 && map != k5kGUniform) {
494  if ((l3Cur <= zero) ||
495  ((convention == kConvLHC && l3Pol != diPol) || (convention == kConvDCS2008 && l3Pol == diPol))) {
496  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN,
497  "Wrong combination for L3/Dipole polarities (%c/%c) for convention %d",
498  l3Pol > 0 ? '+' : '-', diPol > 0 ? '+' : '-', getPolarityConvention());
499  }
500  }
501 
502  if (l3Pol < 0) {
503  sclL3 = -sclL3;
504  }
505  if (diPol < 0) {
506  sclDip = -sclDip;
507  }
508 
509  BeamType_t btype = kNoBeamField;
510  TString btypestr = beamtype;
511  btypestr.ToLower();
512  TPRegexp protonBeam("(proton|p)\\s*-?\\s*\\1");
513  TPRegexp ionBeam("(lead|pb|ion|a|A)\\s*-?\\s*\\1");
514  TPRegexp protonionBeam("(proton|p)\\s*-?\\s*(lead|pb|ion|a|A)");
515  TPRegexp ionprotonBeam("(lead|pb|ion|a|A)\\s*-?\\s*(proton|p)");
516  if (btypestr.Contains(ionBeam)) {
517  btype = kBeamTypeAA;
518  } else if (btypestr.Contains(protonBeam)) {
519  btype = kBeamTypepp;
520  } else if (btypestr.Contains(protonionBeam)) {
521  btype = kBeamTypepA;
522  } else if (btypestr.Contains(ionprotonBeam)) {
523  btype = kBeamTypeAp;
524  } else {
525  FairLogger::GetLogger()->Info(MESSAGE_ORIGIN, "Assume no LHC magnet field for the beam type %s,", beamtype);
526  }
527  char ttl[80];
528  snprintf(ttl, 79, "L3: %+5d Dip: %+4d kA; %s | Polarities in %s convention", (int)TMath::Sign(l3Cur, float(sclL3)),
529  (int)TMath::Sign(diCur, float(sclDip)), uniform ? " Constant" : "",
530  convention == kConvLHC ? "LHC" : "DCS2008");
531  // LHC and DCS08 conventions have opposite dipole polarities
532  if (getPolarityConvention() != convention) {
533  sclDip = -sclDip;
534  }
535 
536  return new MagneticField("MagneticFieldMap", ttl, sclL3, sclDip, map, btype, beamenergy, 2, 10., path);
537 }
